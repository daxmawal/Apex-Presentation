\section{1. Context}
% =========================================================

\begin{frame}[t]{Case study: using APEX on Dyablo}
\small
\begin{itemize}\itemsep0.4em
    \item Dyablo is a Kokkos-based AMR hydrodynamics / MHD application targeting CPU and GPU executions.
    \item APEX is a performance measurement and auto-tuning framework that can plug into Kokkos.
    \item In this presentation, \dyablo{} is used as an example to show:
    \begin{itemize}\itemsep0.2em
        \item how APEX profiles a GPU application
        \item how APEX tunes Kokkos internal execution parameters
    \end{itemize}
    \item Integration branch: \href{https://github.com/daxmawal/Dyablo-apex/tree/friendly_apex_hydroe_patch}{\texttt{daxmawal/Dyablo-apex}, branch \texttt{friendly\_apex\_hydroe\_patch}}.
    \item All Dyablo examples in this presentation are run with \texttt{test\_sod\_3D.ini}.
\end{itemize}
\end{frame}

\begin{frame}[t]{Two complementary roles}
\begin{beamerboxesrounded}[upper=profiler title,lower=profiler body,width=\linewidth,shadow=false]{Profiler}
\small
Aggregate timing and counter statistics.
\vspace{0.5em}
\begin{itemize}\itemsep0.3em
    \item Kokkos regions, kernels, and CUDA runtime calls
    \item NVML GPU metrics
    \item End-of-run performance summary
\end{itemize}
\end{beamerboxesrounded}

\vspace{0.8em}

\begin{beamerboxesrounded}[upper=tuner title,lower=tuner body,width=\linewidth,shadow=false]{Tuner}
\small
Search better Kokkos launch parameters.
\vspace{0.35em}
\begin{itemize}\itemsep0.22em
    \item Handles Kokkos tuning requests
    \item Tries candidate configurations online
    \item Reuses stable configurations and can write them to YAML
    \item If YAML has no final configuration yet, the next Dyablo run retunes that context
\end{itemize}
\end{beamerboxesrounded}
\end{frame}

\begin{lstlisting}[style=terminal,basicstyle=\ttfamily\scriptsize\color{termText}]
$ KOKKOS_TOOLS_LIBS=../apex/install/lib/libapex.so \
> APEX_SCREEN_OUTPUT=1 \
> APEX_KOKKOS_PROFILING_FENCES=1 \
> APEX_ENABLE_CUDA=1 \
> APEX_MONITOR_GPU=1 \
> APEX_CUDA_COUNTERS=1 \
> APEX_CUDA_KERNEL_DETAILS=1 \
> APEX_CUDA_KERNEL_ACTIVITY=1 \

{\color{ceaGreen}\footnotesize Main profiling options}
\footnotesize
\begin{itemize}\itemsep0.06em
    \item \textpdfrender{TextRenderingMode=FillStroke,LineWidth=0.18pt}{\textbf{\texttt{KOKKOS\_TOOLS\_LIBS=...}}} loads \texttt{libapex.so} as the Kokkos tool.
    \item \textpdfrender{TextRenderingMode=FillStroke,LineWidth=0.18pt}{\textbf{\texttt{APEX\_SCREEN\_OUTPUT=1}}} prints the summary in the terminal.
    \item \textpdfrender{TextRenderingMode=FillStroke,LineWidth=0.18pt}{\textbf{\texttt{APEX\_KOKKOS\_PROFILING\_FENCES=1}}} asks Kokkos to fence after kernel launches for more reliable timings.
    \item The last five variables enable CUDA tracing, CUDA counters, kernel details, and NVML metrics.
\end{itemize}
\end{frame}

\section{3. APEX auto-tuning with Dyablo}

\begin{frame}[fragile,t]{APEX auto-tuning command line}
\begin{beamerboxesrounded}[upper=terminal title,lower=terminal body,width=\linewidth,shadow=false]{%
\textcolor{red!70}{\footnotesize\textbullet}\hspace{0.15em}%
\textcolor{yellow!80}{\footnotesize\textbullet}\hspace{0.15em}%
\textcolor{green!70!black}{\footnotesize\textbullet}\hspace{0.55em}%
\texttt{\scriptsize Dyablo with APEX auto-tuning}}
\begin{lstlisting}[style=terminal,basicstyle=\ttfamily\footnotesize\color{termText}]
$ KOKKOS_TOOLS_LIBS=../apex/install/lib/libapex.so \
> APEX_SCREEN_OUTPUT=1 \
> APEX_KOKKOS_PROFILING_FENCES=1 \
> KOKKOS_TUNE_INTERNALS=1 \
> APEX_KOKKOS_TUNING=1 \
> APEX_KOKKOS_TUNING_POLICY=simulated_annealing \
> APEX_KOKKOS_TUNING_WINDOW=5 \
> APEX_KOKKOS_VERBOSE=1 \
> ./dyablo test_sod_3D.ini
\end{lstlisting}
\end{beamerboxesrounded}

{\color{ceaGreen}\scriptsize Main tuning options}
\scriptsize
\begin{itemize}\itemsep0.02em
    \item \textpdfrender{TextRenderingMode=FillStroke,LineWidth=0.18pt}{\textbf{\texttt{KOKKOS\_TUNE\_INTERNALS=1}}} asks Kokkos to expose internal tuners.
    \item \textpdfrender{TextRenderingMode=FillStroke,LineWidth=0.18pt}{\textbf{\texttt{APEX\_KOKKOS\_TUNING=1}}} enables the APEX tuner.
    \item \textpdfrender{TextRenderingMode=FillStroke,LineWidth=0.18pt}{\textbf{\texttt{APEX\_KOKKOS\_TUNING\_POLICY=...}}} selects the search strategy.
    \item \textpdfrender{TextRenderingMode=FillStroke,LineWidth=0.18pt}{\textbf{\texttt{APEX\_KOKKOS\_TUNING\_WINDOW=5}}} requires at least 5 observations for one context before APEX triggers a tuning update.
\end{itemize}
\end{frame}

\begin{frame}[t]{Same APEX context key, different workloads}
\scriptsize
{\color{ceaGreen}\textbf{APEX distinguishes contexts with \texttt{kernel\_name}, \texttt{kernel\_type}, and \texttt{tree\_node}. It does not distinguish them with the current kernel size.}}

\vspace{0.18em}
\begin{columns}[T,onlytextwidth]
\column{0.35\textwidth}
\begin{beamerboxesrounded}[upper=tuner title,lower=tuner body,width=\linewidth,shadow=false]{APEX context key}
\tiny
\begin{itemize}\itemsep0.18em
    \item \texttt{kernel\_name}
    \item \texttt{kernel\_type}
    \item current APEX \texttt{tree\_node}
\end{itemize}
\end{beamerboxesrounded}

\vspace{0.22em}
\begin{beamerboxesrounded}[upper=profiler title,lower=profiler body,width=\linewidth,shadow=false]{Not in the context key}
\tiny
\begin{itemize}\itemsep0.18em
    \item \texttt{nbOcts}
    \item current work distribution
    \item runtime regime changes
\end{itemize}
\end{beamerboxesrounded}

\column{0.62\textwidth}
\begin{beamerboxesrounded}[upper=terminal title,lower=terminal body,width=\linewidth,shadow=false]{What this means in Dyablo}
\tiny
\begin{itemize}\itemsep0.2em
    \item Two calls to \texttt{HydroE::Patch} with different \texttt{number of octants} still hit the same APEX context.
    \item Best \textcolor{green!70!black}{\texttt{team\_size}} and \textcolor{green!70!black}{\texttt{vector\_length}} can change with size, backend, GPU, and workload.
    \item One YAML entry can therefore mix several regimes and end up as a compromise.
    \item If one run stops with \texttt{Converged: false}, the next run does not resume that unfinished search, it starts tuning that context again and may revisit similar candidates.
    \item With short runs, this context may keep getting retuned without ever settling on one stable setting.
\end{itemize}
\end{beamerboxesrounded}
\end{columns}
\end{frame}

  for (int pass = 0; pass < tuning_passes; ++pass) {
    restore_reference_inputs();
    reset_reference_outputs();

  }
  restore_reference_inputs();
  reset_reference_outputs();

\vspace{0.1em}
\scriptsize
\begin{itemize}\itemsep0.08em
    \item Splitting \texttt{HydroE::Patch} into several APEX labels is not enough, APEX still needs repeated visits to each context.
    \item One short Dyablo pass per label is often too short, so the search still ends with \texttt{Converged: false}.
    \item Each tuning pass must restart from the same inputs and outputs, otherwise APEX measures a changing state.
\end{itemize}
